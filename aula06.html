function nomeDaFuncao (parametroA, parametroB) {
    var resultado = parametroA * parametroB;   // comando executado
    return resultado;                          // função tem um retorno

    //não pode colocar nenhum comando após o retorno
}

var x = nomeDaFuncao(4, 3);        // A função é chamada, e o valor retornado é armazenado em x
