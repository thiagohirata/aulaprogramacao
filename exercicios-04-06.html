function desenharBola(x, y) {
						var c = gerarCorAleatoria();

	}
	function desenharCruz(x, y) {
		var c = gerarCorAleatoria();
		stroke(c.red, c.green, c.blue);
	line(50 + x * 150, 50 + y * 150, 150 + x * 150, 150 + y * 150);
	line(150 + x * 150, 50 + y * 150, 50 + x * 150, 150 + y * 150);
	}
	function desenharTabuleiro() {
		var c = gerarCorAleatoria();
		stroke(c.red, c.green, c.blue);
	  line(175, 25, 175, 475);
		c = gerarCorAleatoria();
		stroke(c.red, c.green, c.blue);
		line(325, 25, 325, 475);
		c = gerarCorAleatoria();
		stroke(c.red, c.green, c.blue);
		line(25, 175, 475, 175);
		c = gerarCorAleatoria();
		stroke(c.red, c.green, c.blue);
		line(25, 325, 475, 325);
	}
	function gerarCorAleatoria() {
	  var x = {   red:    Math.random() * 255,
	              green:  Math.random() * 255,
	              blue:   Math.random() * 255};
		return x;
	}
