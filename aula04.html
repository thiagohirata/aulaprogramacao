Exercícios - Escreva o código que tenha a mesma saída

var WIDTH = 500; var HEIGHT = 500;
size(WIDTH, HEIGHT);
var red = 0;
var blue = 255;
var green = 0;
var i = 0;
while (i < 255) {
  red  += 1;
  blue -= 1;
  fill(red, green, blue);
  ellipse(10 + (i % 25) * 20, 10 + Math.floor(i / 25) * 20, 20, 20);
  i++;
}
